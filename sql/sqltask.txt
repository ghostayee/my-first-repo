=>wildcards(%,_)
%=> single,multiple or none characters
_=single character 

display customers with letter m in the customername 

Select * from customers where customername like 'm%';

		order by
=>Keyword used to sort or arrange results 
 using a certain column

		group by
=>Used to group rows that have the same values
=>Mostly used with agregate function(sum,count)

total number of order details for each product
SELECT productid,count(*) from orderdetails group by productid;

total number of orders for each customer 
SELECT customerid ,count(*) FROM orders group by customerid;
